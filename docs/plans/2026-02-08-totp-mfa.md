# TOTP-Based MFA Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Add optional TOTP-based multi-factor authentication with backup codes so users can protect their accounts with a second factor.

**Architecture:** Two new database tables (`user_mfa`, `user_mfa_backup_codes`) store encrypted TOTP secrets and hashed backup codes. The login flow gains a two-step mode: after password verification, MFA-enabled users are redirected to `/login/verify-mfa` with a temporary pending-auth session. The security settings page (`/security`) wires its existing `SecurityMfaCard` shell to a setup wizard modal. All crypto uses Web Crypto API (PBKDF2 for backup code hashing, AES-GCM for secret encryption) for cross-runtime compatibility.

**Tech Stack:** `@oslojs/otp` (TOTP), `qrcode` (QR generation), Web Crypto API (encryption/hashing), Drizzle ORM (dual SQLite/PostgreSQL), Astro components, TypeScript client scripts.

```typescript
/**
 * MFA Cryptographic Utilities
 *
 * Provides encryption/decryption for TOTP secrets (AES-256-GCM)
 * and hashing/verification for backup codes (PBKDF2-SHA256).
 *
 * All operations use Web Crypto API for cross-runtime compatibility
 * (Node.js, Bun, Cloudflare Workers).
 */

const BACKUP_CODE_LENGTH = 8; // 4+4 with dash
const BACKUP_CODE_COUNT = 10;
const PBKDF2_ITERATIONS = 100_000;

/**
 * Encrypt a TOTP secret using AES-256-GCM.
 *
 * Format: base64(iv:ciphertext) where iv is 12 bytes.
 *
 * @param secret - Plaintext TOTP secret (base32-encoded string)
 * @param encryptionKey - Base64-encoded 32-byte encryption key
 * @returns Base64-encoded encrypted string
 */
export async function encryptTotpSecret(secret: string, encryptionKey: string): Promise<string> {
  const keyBytes = base64ToBytes(encryptionKey);
  const key = await crypto.subtle.importKey('raw', keyBytes, 'AES-GCM', false, ['encrypt']);

  const iv = crypto.getRandomValues(new Uint8Array(12));
  const plaintext = new TextEncoder().encode(secret);

  const ciphertext = await crypto.subtle.encrypt({ name: 'AES-GCM', iv }, key, plaintext);

  // Concatenate iv + ciphertext, then base64 encode
  const combined = new Uint8Array(iv.length + ciphertext.byteLength);
  combined.set(iv, 0);
  combined.set(new Uint8Array(ciphertext), iv.length);

  return bytesToBase64(combined);
}

/**
 * Decrypt a TOTP secret encrypted with AES-256-GCM.
 *
 * @param encrypted - Base64-encoded encrypted string (iv:ciphertext)
 * @param encryptionKey - Base64-encoded 32-byte encryption key
 * @returns Plaintext TOTP secret
 */
export async function decryptTotpSecret(encrypted: string, encryptionKey: string): Promise<string> {
  const keyBytes = base64ToBytes(encryptionKey);
  const key = await crypto.subtle.importKey('raw', keyBytes, 'AES-GCM', false, ['decrypt']);

  const combined = base64ToBytes(encrypted);
  const iv = combined.slice(0, 12);
  const ciphertext = combined.slice(12);

  const plaintext = await crypto.subtle.decrypt({ name: 'AES-GCM', iv }, key, ciphertext);

  return new TextDecoder().decode(plaintext);
}

/**
 * Generate 10 single-use backup codes in XXXX-XXXX format.
 *
 * Each code is 8 alphanumeric uppercase characters with a dash separator.
 *
 * @returns Array of 10 plaintext backup codes
 */
export function generateBackupCodes(): string[] {
  const chars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789';
  const codes: string[] = [];
  const seen = new Set<string>();

  while (codes.length < BACKUP_CODE_COUNT) {
    const bytes = crypto.getRandomValues(new Uint8Array(BACKUP_CODE_LENGTH));
    const code = Array.from(bytes)
      .map((b) => chars[b % chars.length])
      .join('');
    const formatted = `${code.slice(0, 4)}-${code.slice(4)}`;

    if (!seen.has(formatted)) {
      seen.add(formatted);
      codes.push(formatted);
    }
  }

  return codes;
}

/**
 * Hash a backup code using PBKDF2-SHA256.
 *
 * Format: base64(salt):base64(hash)
 *
 * @param code - Plaintext backup code (case-insensitive)
 * @returns Hashed backup code string
 */
export async function hashBackupCode(code: string): Promise<string> {
  const normalizedCode = code.toUpperCase().trim();
  const salt = crypto.getRandomValues(new Uint8Array(16));
  const keyMaterial = await crypto.subtle.importKey(
    'raw',
    new TextEncoder().encode(normalizedCode),
    'PBKDF2',
    false,
    ['deriveBits']
  );

  const hash = await crypto.subtle.deriveBits(
    { name: 'PBKDF2', salt, iterations: PBKDF2_ITERATIONS, hash: 'SHA-256' },
    keyMaterial,
    256
  );

  return `${bytesToBase64(salt)}:${bytesToBase64(new Uint8Array(hash))}`;
}

/**
 * Verify a backup code against its hash.
 *
 * @param code - Plaintext backup code to verify (case-insensitive)
 * @param storedHash - Stored hash string (salt:hash format)
 * @returns True if code matches
 */
export async function verifyBackupCode(code: string, storedHash: string): Promise<boolean> {
  const normalizedCode = code.toUpperCase().trim();
  const [saltB64, hashB64] = storedHash.split(':');
  if (!saltB64 || !hashB64) return false;

  const salt = base64ToBytes(saltB64);
  const expectedHash = base64ToBytes(hashB64);

  const keyMaterial = await crypto.subtle.importKey(
    'raw',
    new TextEncoder().encode(normalizedCode),
    'PBKDF2',
    false,
    ['deriveBits']
  );

  const actualHash = new Uint8Array(
    await crypto.subtle.deriveBits(
      { name: 'PBKDF2', salt, iterations: PBKDF2_ITERATIONS, hash: 'SHA-256' },
      keyMaterial,
      256
    )
  );

  // Constant-time comparison
  if (actualHash.length !== expectedHash.length) return false;
  let diff = 0;
  for (let i = 0; i < actualHash.length; i++) {
    diff |= actualHash[i] ^ expectedHash[i];
  }
  return diff === 0;
}

// --- Base64 helpers (cross-runtime) ---

function bytesToBase64(bytes: Uint8Array): string {
  let binary = '';
  for (let i = 0; i < bytes.length; i++) {
    binary += String.fromCharCode(bytes[i]);
  }
  return btoa(binary);
}

function base64ToBytes(b64: string): Uint8Array {
  const binary = atob(b64);
  const bytes = new Uint8Array(binary.length);
  for (let i = 0; i < binary.length; i++) {
    bytes[i] = binary.charCodeAt(i);
  }
  return bytes;
}
```

**Step 2: Implement MFA service**

Create `src/services/mfa.service.ts`:

```typescript
/**
 * MFA Service
 *
 * Manages TOTP-based multi-factor authentication lifecycle:
 * setup, verification, backup codes, and disabling.
 */

import { eq, and, isNull } from 'drizzle-orm';
import { nanoid } from 'nanoid';
import { createTOTPKeyURI, TOTPController } from '@oslojs/otp';
import { createLogger } from '@/lib/logger';
import { getActiveSchema } from '@/db';
import type { IDatabase } from '@/db';
import {
  encryptTotpSecret,
  decryptTotpSecret,
  generateBackupCodes,
  hashBackupCode,
  verifyBackupCode as verifyBackupCodeHash,
} from '@/lib/auth/mfa-crypto';

const log = createLogger('mfa');

const MFA_ISSUER = 'allowealth';
const TOTP_PERIOD_SECONDS = 30;
const TOTP_DIGITS = 6;

/**
 * Get the MFA encryption key from environment.
 * Reuses EMAIL_ENCRYPTION_KEY since it's the same requirement (32-byte base64 key).
 * A dedicated MFA_ENCRYPTION_KEY can be added later if separation is desired.
 */
function getEncryptionKey(): string {
  const key = import.meta.env.EMAIL_ENCRYPTION_KEY;
  if (!key || key === 'your-32-byte-key-base64-encoded') {
    throw new Error('EMAIL_ENCRYPTION_KEY must be set for MFA encryption');
  }
  return key;
}

export interface MfaStatus {
  enabled: boolean;
  hasBackupCodes: boolean;
  backupCodesRemaining: number;
}

export interface MfaSetupResult {
  qrCodeDataUrl: string;
  manualEntryCode: string;
}

export class MfaService {
  private db: IDatabase;
  private schema = getActiveSchema();

  constructor(db: IDatabase) {
    this.db = db;
  }

  /**
   * Initialize MFA setup for a user.
   *
   * Generates a TOTP secret, encrypts it, and stores it with mfa_enabled=false.
   * Returns QR code data URL and manual entry code for the user to scan.
   *
   * If a pending (not-yet-enabled) setup exists, it is overwritten.
   */
  async initSetup(userId: string, userEmail: string): Promise<MfaSetupResult> {
    const encryptionKey = getEncryptionKey();

    // Generate 20-byte random secret (160 bits, RFC 4226 recommended minimum)
    const secretBytes = crypto.getRandomValues(new Uint8Array(20));

    // Create TOTP key URI for QR code
    const totpUri = createTOTPKeyURI(MFA_ISSUER, userEmail, secretBytes, TOTP_PERIOD_SECONDS, TOTP_DIGITS);

    // Base32 encode the secret for manual entry
    const manualEntryCode = base32Encode(secretBytes);

    // Encrypt the secret for storage
    const encryptedSecret = await encryptTotpSecret(manualEntryCode, encryptionKey);

    // Check for existing MFA record
    const existing = await this.db.query.userMfa.findFirst({
      where: eq(this.schema.userMfa.user_id, userId),
    });

    if (existing && existing.mfa_enabled) {
      throw new Error('MFA is already enabled. Disable it first to re-setup.');
    }

    if (existing) {
      // Overwrite pending setup
      await this.db
        .update(this.schema.userMfa)
        .set({ totp_secret: encryptedSecret, updated_at: new Date() })
        .where(eq(this.schema.userMfa.id, existing.id));
    } else {
      await this.db.insert(this.schema.userMfa).values({
        id: nanoid(),
        user_id: userId,
        mfa_enabled: false,
        totp_secret: encryptedSecret,
      });
    }

    // Generate QR code as data URL
    const QRCode = await import('qrcode');
    const qrCodeDataUrl = await QRCode.toDataURL(totpUri);

    log.info(`MFA setup initiated for user ${userId}`);

    return { qrCodeDataUrl, manualEntryCode };
  }

  /**
   * Verify initial setup by checking a TOTP code and enable MFA.
   *
   * @returns Plaintext backup codes (display once, then discard)
   */
  async verifySetup(userId: string, totpCode: string): Promise<string[]> {
    const mfaRecord = await this.getMfaRecord(userId);
    if (!mfaRecord) {
      throw new Error('No MFA setup found. Start setup first.');
    }
    if (mfaRecord.mfa_enabled) {
      throw new Error('MFA is already enabled.');
    }

    // Decrypt and verify TOTP code
    const isValid = await this.validateTotpCode(mfaRecord.totp_secret, totpCode);
    if (!isValid) {
      throw new Error('Invalid verification code. Please try again.');
    }

    // Enable MFA
    await this.db
      .update(this.schema.userMfa)
      .set({ mfa_enabled: true, updated_at: new Date() })
      .where(eq(this.schema.userMfa.id, mfaRecord.id));

    // Generate and store backup codes
    const codes = await this.generateAndStoreBackupCodes(mfaRecord.id);

    log.info(`MFA enabled for user ${userId}`);

    return codes;
  }

  /**
   * Get MFA status for a user.
   */
  async getStatus(userId: string): Promise<MfaStatus> {
    const mfaRecord = await this.db.query.userMfa.findFirst({
      where: eq(this.schema.userMfa.user_id, userId),
    });

    if (!mfaRecord || !mfaRecord.mfa_enabled) {
      return { enabled: false, hasBackupCodes: false, backupCodesRemaining: 0 };
    }

    const backupCodes = await this.db.query.userMfaBackupCodes.findMany({
      where: and(
        eq(this.schema.userMfaBackupCodes.user_mfa_id, mfaRecord.id),
        isNull(this.schema.userMfaBackupCodes.used_at)
      ),
    });

    return {
      enabled: true,
      hasBackupCodes: backupCodes.length > 0,
      backupCodesRemaining: backupCodes.length,
    };
  }

  /**
   * Check if MFA is enabled for a user. Used during login flow.
   */
  async isMfaEnabled(userId: string): Promise<boolean> {
    const mfaRecord = await this.db.query.userMfa.findFirst({
      where: eq(this.schema.userMfa.user_id, userId),
    });
    return mfaRecord?.mfa_enabled === true;
  }

  /**
   * Verify a TOTP code during login.
   */
  async verifyTotp(userId: string, code: string): Promise<boolean> {
    const mfaRecord = await this.getMfaRecord(userId);
    if (!mfaRecord || !mfaRecord.mfa_enabled) {
      return false;
    }
    return this.validateTotpCode(mfaRecord.totp_secret, code);
  }

  /**
   * Verify and consume a backup code during login.
   *
   * @returns true if code is valid and was consumed
   */
  async verifyAndConsumeBackupCode(userId: string, code: string): Promise<boolean> {
    const mfaRecord = await this.getMfaRecord(userId);
    if (!mfaRecord || !mfaRecord.mfa_enabled) {
      return false;
    }

    // Get unused backup codes
    const backupCodes = await this.db.query.userMfaBackupCodes.findMany({
      where: and(
        eq(this.schema.userMfaBackupCodes.user_mfa_id, mfaRecord.id),
        isNull(this.schema.userMfaBackupCodes.used_at)
      ),
    });

    // Check each code (must iterate because codes are hashed)
    for (const stored of backupCodes) {
      const isValid = await verifyBackupCodeHash(code, stored.code_hash);
      if (isValid) {
        // Mark as used
        await this.db
          .update(this.schema.userMfaBackupCodes)
          .set({ used_at: new Date() })
          .where(eq(this.schema.userMfaBackupCodes.id, stored.id));
        log.info(`Backup code consumed for user ${userId}`);
        return true;
      }
    }

    return false;
  }

  /**
   * Disable MFA for a user. Requires a valid TOTP or backup code.
   */
  async disable(userId: string, code: string): Promise<void> {
    const isTotp = await this.verifyTotp(userId, code);
    const isBackup = !isTotp && (await this.verifyAndConsumeBackupCode(userId, code));

    if (!isTotp && !isBackup) {
      throw new Error('Invalid verification code.');
    }

    const mfaRecord = await this.getMfaRecord(userId);
    if (!mfaRecord) {
      throw new Error('MFA is not configured.');
    }

    // Delete backup codes first (FK constraint)
    await this.db
      .delete(this.schema.userMfaBackupCodes)
      .where(eq(this.schema.userMfaBackupCodes.user_mfa_id, mfaRecord.id));

    // Delete MFA record
    await this.db.delete(this.schema.userMfa).where(eq(this.schema.userMfa.id, mfaRecord.id));

    log.info(`MFA disabled for user ${userId}`);
  }

  /**
   * Regenerate backup codes. Requires a valid TOTP code.
   *
   * @returns New plaintext backup codes
   */
  async regenerateBackupCodes(userId: string, totpCode: string): Promise<string[]> {
    const isValid = await this.verifyTotp(userId, totpCode);
    if (!isValid) {
      throw new Error('Invalid verification code.');
    }

    const mfaRecord = await this.getMfaRecord(userId);
    if (!mfaRecord || !mfaRecord.mfa_enabled) {
      throw new Error('MFA is not enabled.');
    }

    // Delete old backup codes
    await this.db
      .delete(this.schema.userMfaBackupCodes)
      .where(eq(this.schema.userMfaBackupCodes.user_mfa_id, mfaRecord.id));

    // Generate new codes
    const codes = await this.generateAndStoreBackupCodes(mfaRecord.id);

    log.info(`Backup codes regenerated for user ${userId}`);

    return codes;
  }

  // --- Private helpers ---

  private async getMfaRecord(userId: string) {
    return this.db.query.userMfa.findFirst({
      where: eq(this.schema.userMfa.user_id, userId),
    });
  }

  private async validateTotpCode(encryptedSecret: string, code: string): Promise<boolean> {
    const encryptionKey = getEncryptionKey();
    const base32Secret = await decryptTotpSecret(encryptedSecret, encryptionKey);
    const secretBytes = base32Decode(base32Secret);

    const totp = new TOTPController();
    // Check current time window and ±1 window (90 seconds total)
    return totp.verify(code, secretBytes, TOTP_PERIOD_SECONDS, TOTP_DIGITS);
  }

  private async generateAndStoreBackupCodes(mfaId: string): Promise<string[]> {
    const codes = generateBackupCodes();

    for (const code of codes) {
      const codeHash = await hashBackupCode(code);
      await this.db.insert(this.schema.userMfaBackupCodes).values({
        id: nanoid(),
        user_mfa_id: mfaId,
        code_hash: codeHash,
      });
    }

    return codes;
  }
}

// --- Base32 helpers ---

const BASE32_ALPHABET = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ234567';

function base32Encode(bytes: Uint8Array): string {
  let result = '';
  let buffer = 0;
  let bitsLeft = 0;

  for (const byte of bytes) {
    buffer = (buffer << 8) | byte;
    bitsLeft += 8;
    while (bitsLeft >= 5) {
      bitsLeft -= 5;
      result += BASE32_ALPHABET[(buffer >> bitsLeft) & 0x1f];
    }
  }

  if (bitsLeft > 0) {
    result += BASE32_ALPHABET[(buffer << (5 - bitsLeft)) & 0x1f];
  }

  return result;
}

function base32Decode(encoded: string): Uint8Array {
  const cleaned = encoded.toUpperCase().replace(/=+$/, '');
  const output: number[] = [];
  let buffer = 0;
  let bitsLeft = 0;

  for (const char of cleaned) {
    const val = BASE32_ALPHABET.indexOf(char);
    if (val === -1) continue;
    buffer = (buffer << 5) | val;
    bitsLeft += 5;
    if (bitsLeft >= 8) {
      bitsLeft -= 8;
      output.push((buffer >> bitsLeft) & 0xff);
    }
  }

  return new Uint8Array(output);
}
```

```typescript
/**
 * POST /api/auth/login/verify-mfa
 *
 * Verify MFA code during login flow.
 * Expects a pending MFA session (stored in cookie).
 */

import type { APIRoute } from 'astro';
import { mfaService } from '@/services';
import { auth } from '@/lib/auth/lucia';
import {
  createErrorResponseResponse,
  createSuccessResponse,
  STANDARD_RESPONSE_HEADERS,
} from '@/types/api';
import { logError } from '@/lib/utils';
import {
  checkRateLimit,
  createRateLimitResponse,
  applyRateLimitHeaders,
  RATE_LIMIT_PRESETS,
} from '@/lib/rate-limit';

export const prerender = false;

const MFA_PENDING_COOKIE = 'mfa_pending';

export const POST: APIRoute = async ({ request, cookies, clientAddress }) => {
  // Rate limit: 5 attempts per 15 minutes
  const rateLimitResult = checkRateLimit(request, RATE_LIMIT_PRESETS.auth, clientAddress);
  if (!rateLimitResult.allowed) {
    return createRateLimitResponse(rateLimitResult, 'Too many MFA attempts. Please wait.');
  }

  try {
    const pendingData = cookies.get(MFA_PENDING_COOKIE)?.value;
    if (!pendingData) {
      return createErrorResponseResponse('NO_PENDING_MFA', 'No pending MFA verification', 400);
    }

    const { userId } = JSON.parse(pendingData);
    if (!userId) {
      return createErrorResponseResponse('INVALID_STATE', 'Invalid MFA session', 400);
    }

    const { code, isBackupCode } = await request.json();

    if (!code || typeof code !== 'string') {
      return createErrorResponseResponse('INVALID_INPUT', 'Verification code is required', 400);
    }

    let isValid = false;

    if (isBackupCode) {
      isValid = await mfaService.verifyAndConsumeBackupCode(userId, code.trim());
    } else {
      isValid = await mfaService.verifyTotp(userId, code.trim());
    }

    if (!isValid) {
      const response = createErrorResponseResponse(
        'INVALID_MFA_CODE',
        'Invalid verification code',
        401
      );
      return applyRateLimitHeaders(response, rateLimitResult);
    }

    // MFA verified — create full session
    const session = await auth.createSession(userId, {});
    const sessionCookie = auth.createSessionCookie(session.id);

    // Clear pending MFA cookie
    cookies.delete(MFA_PENDING_COOKIE, { path: '/' });

    const response = new Response(
      JSON.stringify(createSuccessResponse({ verified: true })),
      {
        status: 200,
        headers: {
          ...STANDARD_RESPONSE_HEADERS,
          'Set-Cookie': sessionCookie.serialize(),
        },
      }
    );

    return applyRateLimitHeaders(response, rateLimitResult);
  } catch (error) {
    logError('MFA verification error', error);
    return createErrorResponseResponse('MFA_VERIFY_ERROR', 'Verification failed', 500);
  }
};
```

```astro
      if (!isBackupMode && !/^\d{6}$/.test(code)) {
        showError('Please enter a 6-digit code');
        return;
      }

      // Loading state
      if (buttonText) buttonText.classList.add('hidden');
      if (loadingSpinner) loadingSpinner.classList.remove('hidden');
      if (submitButton) submitButton.disabled = true;

      try {
        const response = await fetch(apiVerifyUrl, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          credentials: 'include',
          body: JSON.stringify({ code, isBackupCode: isBackupMode }),
        });

        const result = await response.json();

        if (response.ok && result.success) {
          const returnUrl = redirectTo || '/dashboard';
          window.location.href = returnUrl;
        } else {
          showError(result.error?.message || 'Invalid code. Please try again.');
        }
      } catch (error) {
        showError('An error occurred. Please try again.');
      } finally {
        if (buttonText) buttonText.classList.remove('hidden');
        if (loadingSpinner) loadingSpinner.classList.add('hidden');
        if (submitButton) submitButton.disabled = false;
      }
    });
  }
</script>
```
